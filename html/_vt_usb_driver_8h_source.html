429  virtual vt_bool send_command(vt_byte status[16]
430  , const vt_byte CommandCode
431  , const vt_uint16 SubCommandCode )
432  {
433  WDU_DEVICE_HANDLE hDevice = driver_handle();
434  vt_ulong bufSize = 16;
435  vt_ulong dwBytesTransferred;
436 
437  // Look up the command codes
438  // Define the direction control byte for USB transfer which comprises of a bit pattern representing
439  // command out, command out with data, command in with data generated from the following table:
440  // Bit 7 - Request direction (0 - host to device, 1 - device to host)
441  // Bits 5 & 6 - Request type (0 - standard, 1 - class, 2 - vendor, 3 - reserved)
442  // Bits 0 to 4 - Recipient (0 - device, 1 - interface, 2 - endpoint, 3 - other)
443  //
444  // Thus:
445  // Standard command send with no data transfer = (00000010 == 0x2)
446  // Vendor command send with data send to device = (01000000 == 0x40)
447  // Standard command send with data retreived from device = (10000010 == 0x80)
448  // Vendor command send with data received from device = (11000000 == 0xC0)
449  vt_byte DirectionControl = 0xC0;
450 
451  // Define the standard USB command packet from the input data
452  vt_byte Packet[8];
453  Packet[0] = DirectionControl; // Direction of the transfer and if there is also data
454  Packet[1] = CommandCode; // The command code
455  Packet[2] = (vt_byte)SubCommandCode; // The first half of the sub command code
456  Packet[3] = SubCommandCode >> 8; // The second half of the sub command code
457  Packet[4] = 0; // Unused
458  Packet[5] = 0; // Unused
459  Packet[6] = 0x10; // The first half of the return data size (0 as we are reading from a pipe)
460  Packet[7] = 0; // The second half of the return data size (0 as we are reading from a pipe)
461 
462  // Send the command - debug version to read the debug 16 byte packet
463  memset( status, 0, bufSize*sizeof( vt_byte ) );
464 
465  WDU_TransferDefaultPipe( hDevice // device handle
466  , TRUE // read flag
467  , 0 // there are no options
468  , status
469  , bufSize
470  , &dwBytesTransferred
471  , Packet
472  , TRANSFER_TIMEOUT );
473 
474  if(bufSize != dwBytesTransferred )
475  {
476  BUGOUT( stderr,"Failed to send command to USB device.");
477  return false;
478  }
479  return true;
480  }

584 
585  pListenPipe->fStarted = TRUE;

589

593  BUGOUT( stderr, "Buffer count %d Bytes Requested %ld \n", dwNumBufs, (unsigned long) rxDataSize );
594  }
595  start = clock();
596  DWORD dwBytesTransferred;
597  for (vt_ulong count=0;count<dwNumBufs;count++)
598  {
599 
600  vt_acq_im_type *buf;
601 
602  buf = pData[count];
603 
604  dwError = WDU_Transfer(pListenPipe->hDevice
605  , pListenPipe->dwPipeNum

610  , NULL
611  , TRANSFER_TIMEOUT);
612

615  pListenPipe->dwError = dwError;

617  }
618  }

621 
622  double data_rx = count*rxDataSize;
