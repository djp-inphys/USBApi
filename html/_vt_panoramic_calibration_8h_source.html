264  // smooth
265  //
266  // Produces a smoothed version of the input vector
267  // Returns smoothed version as a pointer it is the calling functions responsibility to remove

269  //
270  CoefType * smooth(CoefType *vec, const vt_ulong length)
271  {
272  CoefType * smth_vec = new CoefType[length]; //larger than require
273 
274  memset( smth_vec, 0 , sizeof( smth_vec[0] )*length );
275 
276  // smooth
277  vt_double sum = 0;
278  vt_long smooth_span = SMOOTH_SPAN;
279  for( vt_ulong pos = smooth_span; pos < length - smooth_span; pos++ )
280  {
281  if (smooth_span == pos) // first pixel
282  {
283  for( vt_long idx = -smooth_span; idx < smooth_span ; idx++)
284  {
285  sum += vec[pos+idx];
286  }
287 
288  // deal with beginning
289  CoefType mn = sum/TOTAL_SPAN;
290  for(idx = 0; idx < smooth_span ; idx++)
291  {
292  smth_vec[idx] = mn;
293  }
294  }
295  else
296  {
297  vt_ulong last_start = pos - smooth_span - 1;
298  vt_ulong new_end = pos + smooth_span;
299 
300  sum -= vec[last_start];
301  sum += vec[new_end];
302  }
303 
304  smth_vec[pos] = sum/TOTAL_SPAN;
305 
306  if (length - smooth_span - 1 == pos) // last pixel
307  {
308  // deal with end - jut tidy up last few elements of the vector
309  CoefType mn = sum/TOTAL_SPAN; // this will fill the last
310  // few entries with the last
311  // valid smoothed entry
312  for(vt_ulong idx = pos+1; idx < length; idx++)
313  {
314  smth_vec[idx] = mn;
315  }
316  }
317  }
318 
319  // deal with the
320  return smth_vec;
321  }
